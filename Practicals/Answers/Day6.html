Practical 1: Basic Routing Setup

Task:

  • Create three components: Home, About, and Contact
  • Set up routing using react-router-dom
  • Show navigation links to switch between pages without refreshing

Expected Output:

  • Clicking "About" shows About component
  • Clicking "Contact" shows Contact component

Solution:

src/
├── App.js
├── components/
│ ├── Home.js
│ ├── About.js
│ └── Contact.js

App.js

import { BrowserRouter as Router, Routes, Route, Link } from 'react-router-dom';
import Home from './components/Home';
import About from './components/About';
import Contact from './components/Contact';

function App() {
  return (
    <Router>
      <nav>
        <Link to="/">Home</Link> | 
        <Link to="/about">About</Link> | 
        <Link to="/contact">Contact</Link>
      </nav>
      
      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/about" element={<About />} />
        <Route path="/contact" element={<Contact />} />
      </Routes>
    </Router>
  );
}

export default App;

components/Home.js

function Home() {
  return <h2>Welcome to our Home Page!</h2>;
}
export default Home;

components/About.js

function About() {
  return <h2>About Our Company</h2>;
}
export default About;

components/Contact.js

function Contact() {
  return <h2>Contact Us Page</h2>;
}
export default Contact;

Practical 4: Navigation Using Link

Task:

  • Replace anchor tags (<a>) with <Link> components

Expected Output:

  • No page reload when navigating between pages

Solution:

Replace all anchor tags like this:

// Bad - causes full page reload
<a href="/about">About</a>

// Good - uses client-side routing
<Link to="/about">About</Link>

Example from App.js navigation:

import { Link } from 'react-router-dom';

// In your component:
<nav>
  <Link to="/">Home</Link> | 
  <Link to="/about">About</Link> | 
  <Link to="/contact">Contact</Link>
</nav>

Practical 5: Create Navbar Component

Task:

  • Create a reusable Navbar component
  • Include it in App.js to appear on all pages

Expected Output:

  • Consistent navigation across all pages

Solution:

src/
├── App.js (updated)
└── components/
└── Navbar.js

components/Navbar.js

import { Link } from 'react-router-dom';

function Navbar() {
  return (
    <nav style={{ padding: '10px', background: '#f0f0f0' }}>
      <Link to="/" style={{ marginRight: '10px' }}>Home</Link>
      <Link to="/about" style={{ marginRight: '10px' }}>About</Link>
      <Link to="/contact" style={{ marginRight: '10px' }}>Contact</Link>
      <Link to="/dashboard">Dashboard</Link>
    </nav>
  );
}
export default Navbar;
