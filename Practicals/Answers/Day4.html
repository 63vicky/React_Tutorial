function FruitList() {
  const fruits = ['Apple', 'Banana', 'Mango'];
  return (
    <ul>
      {fruits.map((fruit, index) => (

ðŸ”¹ Practical 4: Conditional List Rendering

function UserList() {
  const users = ['Ali', 'Pooja']; // or []

  return (
    <div>
      {users.length > 0 ? (
        <ul>
          {users.map((user, index) => (
            <li key={index}>{user}</li>
          ))}
        </ul>
      ) : (
        <p>No users online</p>
      )}
    </div>
  );
}

Key Points:

  • Ternary operator for conditional rendering
  • Checking array length to determine what to show
  • Fallback UI when list is empty

ðŸ”¸ Practical 7: Dynamic Styling with useState 

import { useState } from 'react';

function ColorChanger() {
  const [bgColor, setBgColor] = useState('red');

  const changeColor = () => {
    setBgColor(prev => (prev === 'red' ? 'blue' : 'red'));
  };

  return (
    <div style={{ backgroundColor: bgColor, padding: '20px' }}>
      <button onClick={changeColor}>Change Background</button>
    </div>
  );
}

Key Points:

  • State variable to track current color
  • Inline styles that change based on state
  • Toggle logic using functional update
